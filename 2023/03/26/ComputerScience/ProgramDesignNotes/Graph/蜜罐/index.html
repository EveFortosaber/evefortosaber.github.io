# 题目描述

蜜蜂乔治在采蜂蜜的时候，会把蜜放进 N 个蜜罐中，现在需要把这 N 个不相邻的蜜罐构建一些双向边把它们连接起来。即有 N 个分立的点，同时给出 M 个不同的边，需要你求出这张图的最小生成树。或许你会觉得这种描述有些奇怪，不过使用如此描述完全由于出题人喜欢吃蜂蜜。

# 输入

第一行输入一个 T，表示样例个数。
对于每一个样例，第二行输入 N, M，表示蜜罐数与边的数量。
第三行输入三个数字 s, t, w 表示双向边的两个端点和权重。

# 输出

对于每个样例，输出一个数字 res 表示生成树的权重。如果无法构成生成树，则输出 - 1.

w < 10000

# 解题思路

# 代码
#include <iostream>
#include <vector>
#include <math.h>
using namespace std;
struct vertex//边节点，由一个节点指向节点number，权重为weight
{
    int number;
    int weight;
};
int N,M;//n is the number of vertex points, m is the number of edges
vector<vertex> graph[10000];//graph,以边为主
int visited[10000]={0};//if is visited
int idstances[10000]={0};//the distance between the visited points and the unvisited points
void prime(int node,int num,int &sum)//对于node节点有边graph[node][i]和对应的权值
{
    if(num==N)// when the node is the last node
    {
        return;
    }
    int size = graph[node].size();// all the edges of the node
    for(int i=0;i<size;i++)// 逐个对distance更新
    {
        if(!visited[graph[node][i].number])//graph[node][i].number是node链接的节点标号
        {
            if(idstances[graph[node][i].number]>graph[node][i].weight||idstances[graph[node][i].number]==0){
                idstances[graph[node][i].number] = graph[node][i].weight;//reload idstances that is effected by the edge
            }
        }
    }
    int mini=10000;//to find the minimum distance to the unvisited points
    int nnode;
    for(int i=1;i<=N;i++)
    {
        if(!visited[i])
        {
            mini = min(mini,idstances[i]);
            nnode=i;// go to the new node
        }
    }
    visited[nnode]=1;
    sum+=mini;
    prime(nnode,num+1,sum);// begin with the new node
}


int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        cin>>N>>M;
        for(int i=1;i<=N;i++)
        {
            visited[i]=0;// reset visited
            graph[i].clear();// reset graph
        }
        for(int i=1;i<=M;i++)// load the graph
        {
            int u,v,w;
            cin>>u>>v>>w;
            vertex temp1,temp2;
            temp1.number=u;
            temp1.weight=w;
            temp2.number=v;
            temp2.weight=w;
            graph[u].push_back(temp2);
            graph[v].push_back(temp1);
        }
        int sum=0;
        visited[1]=1;// to start with the first node
        prime(1,1,sum);
        cout<<sum<<endl;
    }
}

# 结语

学习了 Prim 算法，应用贪心思想。
