# 小木棍组正方形：

# 题目描述

有 n 个木棒，需要用上所有木棒，围成一个正方形，如果可以围成正方形，则输出 "yes", 否则输出 "no"。

# 输入

第一行输入一个整数 T 表示样例个数。
对于每个样例，第一行输入一个整数 N 表示木棍的个数，第二行输入 N 个数字表示木棒的长度。

# 输出

对于每个样例，如果可以则输出 "yes", 否则输出 "no"。

# 样例输入 复制

# 样例输出 复制

# 提示

N <= 20

# 解题思路：

使用 DFS 算法，并减去一些枝干

# 代码如下
#include <iostream>
#include <algorithm>

using namespace std;
int sticks[38]{0};//建立木棍数组

int side, groupnum, n;
// sum表示目前木棍长度和，num表示已凑出的边数，pos表示使用的木棍，flag表示是否已使用
bool DFS(int sum, int number, int pos,int * flag)
{
    //凑到3条边即可
    if (number == 3)
        return true;
    // 从pos到n-1
    for(int i=pos;i<n;i++)
    {
        // 如果大于边长舍去这种可能
        if(sum+sticks[pos]>side){
            continue;
        }
        // 如果被使用了，跳过
        if(flag[i]==1)
        {
            continue;
        }
        // 恰好找到了
        if(sum+sticks[i]==side)
        {
            // 找到一条边，重头再找下一条边 sum，pos 归零，num+1，但flag不动用于记录是否被使用
            //（如果木棍被用于组成上一条边此时其对应的flag应为1
            if(DFS(0,number+1,0,flag)) return true;
        }
        // 不足
        else
        {
            // 取出这一根木棍：sum=sum+sticks[i] （两个sum不等价）
            if(DFS(sum+sticks[i],number,0,flag)) return true;
        }
        // 回溯，（这种可能不行（此路不通（把用了的木棍归还（则flag归零
        flag[i]=0;
    }
    return false;
}

int main()
{

    cin >> groupnum;//输入测试案例数
    while (groupnum--)
    {
        int flag[38]{0};// 用于标记是否使用过
        cin >> n;// 木棍数量
        int s = 0; // 木棍总长度
        //获取木棍长度，及总长
        for (int i = 0; i < n; i++) 
        {
            cin >> sticks[i];
            s += sticks[i];
        }
        // 如果总长不能被4整除，舍去
        if (s % 4 != 0)
        {
            cout << "no" << endl;
            continue;
        }
        // 边长
        side = s / 4;
        // 排序，便于搜索
        sort(sticks, sticks + n, greater<int>());
        // 如果最长的大于边长舍去
        if (sticks[0] > side)
        {
            cout << "no" << endl;
            continue;
        }
        //搜索
        if (DFS(0, 0, 0,flag))
            cout << "yes" << endl;
        else 
            cout<<"no"<<endl;
    }

# 结语：

DFS 模板

int check(参数)
{
    if(满足条件)
        return 1;
    return 0;
}
 
void dfs(int step)
{
        判断边界
        {
            相应操作
        }
        尝试每一种可能
        {
               满足check条件
               标记
               继续下一步dfs(step+1)
               恢复初始状态（回溯的时候要用到）
        }
}
